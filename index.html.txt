  <script>
    function copyText(){
      const text = 'Słowa mogą ranić — ale też leczyć. Wybieraj je tak, by zostawić ślad, z którego można być dumnym.';
      navigator.clipboard?.writeText(text).then(()=> {
        alert('Cytat skopiowany do schowka!');
      }).catch(()=> alert('Kopiowanie nie powiodło się — skopiuj ręcznie.'));
    }

    function toggleTheme(){
      const dark = document.documentElement.style.getPropertyValue('--bg') === '#fbfbfc';
      if(!dark){
        document.documentElement.style.setProperty('--bg','#10121a');
        document.body.style.background='linear-gradient(180deg,#0b1220 0%, #0f1724 100%)';
        document.body.style.color='#e6eef8';
      } else {
        document.documentElement.style.setProperty('--bg','#fbfbfc');
        document.body.style.background='linear-gradient(180deg,#f6f8ff 0%,#fbfbfc 100%)';
        document.body.style.color='#0f1724';
      }
    }

    function shareNote(){
      const url = location.href;
      if(navigator.share){
        navigator.share({title: document.title, text: 'Twoje słowa — Twój ślad', url}).then(()=> {
          document.getElementById('shareMsg').textContent = 'Udostępniono.';
        }).catch(()=> {
          document.getElementById('shareMsg').textContent = 'Udostępnianie przerwane.';
        });
      } else {
        navigator.clipboard?.writeText(url).then(()=> {
          document.getElementById('shareMsg').textContent = 'Link skopiowany do schowka — wklej go do wiadomości.';
        }).catch(()=> {
          document.getElementById('shareMsg').textContent = 'Skopiuj adres strony ręcznie.';
        });
      }
    }
  </script>
